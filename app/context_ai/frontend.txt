import { Stack } from 'expo-router';
import React from 'react';
import { AuthProvider } from './context/AuthContext';


export default function RootLayout() {
return (
  <AuthProvider>
    <Stack screenOptions={{ headerShown: false }}>
      {/* La primera pantalla del Stack será el grupo de pestañas */}
      <Stack.Screen name="(tabs)" options={{ headerShown: false }} />

      {/* Aquí van las pantallas que están en el Stack principal y no en las pestañas */}
      <Stack.Screen name="register" options={{ title: 'Registro' }} />
      <Stack.Screen name="course" options={{ title: 'Curso' }} />
      <Stack.Screen name="members" options={{ title: 'Miembros' }} />
      <Stack.Screen name="talk" options={{ title: 'Charla' }} />
      <Stack.Screen name="profile" options={{ title: 'Perfil' }} />
      <Stack.Screen name="joinCourse" options={{ title: 'Unirse a Curso' }} />
      <Stack.Screen name="actualCourse" options={{ title: 'Curso Actual' }} />
      <Stack.Screen name="courseIn" options={{ title: 'Curso Participado' }} />

      {/* Manejo de autenticación: Si el usuario no está logueado, se muestra la pantalla de login */}
      {/* Esto es más complejo en Expo Router y generalmente se maneja con un Context o Redux */}
      {/* Por ahora, asumiremos que 'login' es una ruta normal, pero luego veremos cómo proteger rutas */}
      <Stack.Screen name="login" options={{ headerShown: false }} />
    </Stack> 
  </AuthProvider>
);
}
